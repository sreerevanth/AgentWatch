command: agentwatch confidence
    help: [bold]Score[/bold] execution confidence and detect anomalies for a session.
    short_help: 
    param session_file kind=TyperArgument type=path required=True default=None flags=session_file help='Path to session JSON file'

command: agentwatch replay
    help: [bold]Replay[/bold] a captured session step-by-step.
    short_help: 
    param failure_only kind=TyperOption type=boolean required=False default=False flags=--failures,-f help='Show only failure points'
    param from_step kind=TyperOption type=int required=False default=0 flags=--from help='Start from step N'
    param session_file kind=TyperArgument type=path required=True default=None flags=session_file help='Path to session JSON file'
    param show_all kind=TyperOption type=boolean required=False default=False flags=--all help='Show all events including metadata'
    param speed kind=TyperOption type=str required=False default='instant' flags=--speed,-s help='instant|fast|normal|slow'
    param to_step kind=TyperOption type=int required=False default=None flags=--to help='End at step N'

command: agentwatch safety
    help: [bold]Score[/bold] the risk level of a shell command.
    short_help: 
    param command kind=TyperArgument type=str required=True default=None flags=command help='Command to risk-score'
    param verbose kind=TyperOption type=boolean required=False default=False flags=--verbose,-v help=''
